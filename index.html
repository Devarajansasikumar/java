10.BioComputing 
                public class biocom
                {
                    public static int score_match(String subject,String query,int subject_start,int query_start,int length)
                    { 
                        int score = 0;
                        for(int i=0;i < length;i++)
                        {
                            char subject_base = subject.charAt(subject_start+i); 
                            char query_base = query.charAt(query_start+i); 
                            if(subject_base == query_base)
                            {
                                score=score+1;
                            }
                            else
                            {
                                score=score-1;
                            }
                        }
                        return score;
                    }
                    public static void main(String[] args) 
                    {
                        String one_sequence = "actgatcgattgatcgatcgatcg"; 
                        String another_sequence = "tttagatcgatctttgatc";
                        System.out.println(score_match(one_sequence,another_sequence,7,4,8)); 
                        System.out.println(score_match(one_sequence, another_sequence, 7, 4, 4));
                        System.out.println(score_match(one_sequence, another_sequence, 7, 4, 12));
                        System.out.println(score_match(one_sequence, another_sequence, 10,1,5));
                    }
                }
